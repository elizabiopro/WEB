CREATE DATABASE avg_rating_db;
USE avg_rating_db;

CREATE TABLE rating_sets (
    id INT AUTO_INCREMENT PRIMARY KEY,
    r1 INT,
    r2 INT,
    r3 INT,
    r4 INT,
    r5 INT,
    average FLOAT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


select * from rating_sets;
